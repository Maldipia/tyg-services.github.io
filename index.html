Technical Implementation

Sample of our Google Apps Script automation

function sendInvoiceEmail(rowData) {
  // Auto-generate invoice from order data
  const invoice = {
    orderId: rowData.orderId,
    customer: rowData.customerName,
    items: parseItems(rowData.orderDetails),
    total: calculateTotal(rowData),
    payment: rowData.paymentMethod
  };
  
  // Create PDF invoice
  const pdf = createPDF(invoice);
  
  // Send personalized email
  MailApp.sendEmail({
    to: rowData.email,
    subject: `Invoice #${invoice.orderId}`,
    htmlBody: generateEmailTemplate(invoice),
    attachments: [pdf]
  });
  
  // Update status in sheet
  updateOrderStatus(rowData.row, 'INVOICE_SENT');
}
